#send_mode=I# #target=AHK Note#
note() 
;#target:Warcraft III#

chat, -----------extended tests----------

; 정의되지 않은 변수에 빈기본값
chat, %unknown|%                ; 

; 기본값 있는 미정의 변수
chat, % unknown|999 %           ; 999

; 변수 안에 또 변수
#x:5#
#y:x+1#
chat, % y %                     ; 6

; 문자열 결합
#name:James#
#greet:Hello, name!#
chat, % greet %                 ; Hello, James!

; 수식 평가 포함된 기본값
chat, % notset|1+2*3 %          ; 7

; 변수 이름 포함된 기본값 (치환 안 될 경우 그대로 출력)
chat, % missing|abc_name %    ; abc_James

; 소수점 처리 테스트
#a:7.1234# #b:2#
chat, % a / b % #dp_mode:fixed#
chat, % a / b % #dp_mode:ceil#
chat, % a / b % #dp_mode:floor#
chat, % a / b % #dp_mode:round2#

; 복잡한 수식
#a:3# #b:4# #c:5#
chat, % a + b * c - (b**2) %    ; 3 + 4*5 - 16 = 7

; 기본값이 빈 문자열일 때
#nickname:|guest#
chat, % nickname %              ; guest

chat, -----------end----------
